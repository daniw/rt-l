\[

\end{array}

\[

\[ G_{YU}(s) = \frac{Y(s)}{U(s)} \]
\[ \left(A_T \cdot s

\[ \frac{Y(s)}{U(s)}

\[ \to \tau = \frac{A_T}{\sqrt{2 \cdot g} \cdot (\bar{\alpha_2} \cdot A_L + \bar{\alpha_{22}} \cdot A_V)} \]
